import React, { useState, useEffect, useMemo } from 'react';
import { initializeApp } from 'firebase/app';
import { 
  getFirestore, 
  collection, 
  doc, 
  setDoc, 
  addDoc, 
  onSnapshot, 
  query, 
  updateDoc, 
  deleteDoc,
  timestamp
} from 'firebase/firestore';
import { 
  getAuth, 
  signInAnonymously, 
  signInWithCustomToken, 
  onAuthStateChanged 
} from 'firebase/auth';
import { 
  Plus, 
  Wallet, 
  Users, 
  BookOpen, 
  Trash2, 
  CheckCircle, 
  XCircle, 
  TrendingUp, 
  TrendingDown,
  ChevronRight,
  ClipboardList,
  Trophy,
  Info,
  Calendar,
  DollarSign
} from 'lucide-react';

// Firebase 配置 (由環境提供)
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'club-manager-2026';

export default function App() {
  const [user, setUser] = useState(null);
  const [activeTab, setActiveTab] = useState('accounting'); 
  const [transactions, setTransactions] = useState([]);
  const [courses, setCourses] = useState([]);
  const [reimbursements, setReimbursements] = useState([]);
  const [loading, setLoading] = useState(true);

  // --- 表單狀態 ---
  const [showAddTx, setShowAddTx] = useState(false);
  const [newTx, setNewTx] = useState({ type: 'expense', amount: '', category: '老師費用', note: '', date: new Date().toISOString().split('T')[0], isEstimated: false });
  
  const [showAddCourse, setShowAddCourse] = useState(false);
  const [newCourse, setNewCourse] = useState({ name: '', date: '', fee: '', description: '' });
  const [selectedCourse, setSelectedCourse] = useState(null);
  const [newStudent, setNewStudent] = useState({ name: '', phone: '' });

  const [showAddReimbursement, setShowAddReimbursement] = useState(false);
  const [newReimbursement, setNewReimbursement] = useState({ month: '3月', target: '社團補助金', amount: '', invoiceOwner: '', note: '' });

  // 1. 認證處理
  useEffect(() => {
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          await signInWithCustomToken(auth, __initial_auth_token);
        } else {
          await signInAnonymously(auth);
        }
      } catch (error) {
        console.error("Auth error:", error);
      }
    };
    initAuth();

    const unsubscribe = onAuthStateChanged(auth, (u) => {
      setUser(u);
    });
    return () => unsubscribe();
  }, []);

  // 2. 資料監聽 (Firestore)
  useEffect(() => {
    if (!user) return;

    // 監聽記帳資料
    const unsubTx = onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'transactions'), (snapshot) => {
      setTransactions(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })).sort((a, b) => new Date(b.date) - new Date(a.date)));
      setLoading(false);
    }, (err) => console.error(err));

    // 監聽課程資料
    const unsubCourse = onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'courses'), (snapshot) => {
      setCourses(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })).sort((a, b) => new Date(b.date) - new Date(a.date)));
    }, (err) => console.error(err));

    // 監聽核銷資料
    const unsubReim = onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'reimbursements'), (snapshot) => {
      setReimbursements(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
    }, (err) => console.error(err));

    return () => {
      unsubTx();
      unsubCourse();
      unsubReim();
    };
  }, [user]);

  // --- 財務邏輯 ---
  const stats = useMemo(() => {
    const income = transactions.filter(t => t.type === 'income' && !t.isEstimated).reduce((sum, t) => sum + Number(t.amount), 0);
    const expense = transactions.filter(t => t.type === 'expense').reduce((sum, t) => sum + Number(t.amount), 0);
    const estIncome = transactions.filter(t => t.type === 'income' && t.isEstimated).reduce((sum, t) => sum + Number(t.amount), 0);
    return { income, expense, balance: income - expense, estIncome };
  }, [transactions]);

  const handleAddTx = async (e) => {
    e.preventDefault();
    if (!user) return;
    try {
      await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'transactions'), {
        ...newTx,
        amount: Number(newTx.amount),
        createdBy: user.uid,
        createdAt: new Date().toISOString()
      });
      setShowAddTx(false);
      setNewTx({ type: 'expense', amount: '', category: '老師費用', note: '', date: new Date().toISOString().split('T')[0], isEstimated: false });
    } catch (err) { console.error(err); }
  };

  // --- 課程邏輯 ---
  const handleRegister = async (e) => {
    e.preventDefault();
    if (!newStudent.name || !selectedCourse || !user) return;
    const student = { id: Date.now().toString(), name: newStudent.name, phone: newStudent.phone, paid: false, registeredAt: new Date().toISOString() };
    const updatedStudents = [...(selectedCourse.students || []), student];
    try {
      await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'courses', selectedCourse.id), { students: updatedStudents });
      setNewStudent({ name: '', phone: '' });
      setSelectedCourse({ ...selectedCourse, students: updatedStudents });
    } catch (err) { console.error(err); }
  };

  const togglePaid = async (courseId, studentId) => {
    if (!user) return;
    const course = courses.find(c => c.id === courseId);
    const updatedStudents = course.students.map(s => s.id === studentId ? { ...s, paid: !s.paid } : s);
    try {
      await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'courses', courseId), { students: updatedStudents });
      if (selectedCourse?.id === courseId) setSelectedCourse({ ...course, students: updatedStudents });
    } catch (err) { console.error(err); }
  };

  // --- 核銷邏輯 ---
  const handleAddReimbursement = async (e) => {
    e.preventDefault();
    if (!user) return;
    try {
      await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'reimbursements'), {
        ...newReimbursement,
        amount: Number(newReimbursement.amount),
        timestamp: new Date().toISOString()
      });
      setShowAddReimbursement(false);
      setNewReimbursement({ month: '3月', target: '社團補助金', amount: '', invoiceOwner: '', note: '' });
    } catch (err) { console.error(err); }
  };

  if (loading && !user) return <div className="flex items-center justify-center h-screen bg-slate-50"><div className="animate-spin rounded-full h-12 w-12 border-b-2 border-indigo-600"></div></div>;

  return (
    <div className="min-h-screen bg-[#F8FAFC] text-slate-900 font-sans pb-24">
      {/* Header */}
      <header className="bg-white border-b sticky top-0 z-30 px-6 py-4 shadow-sm backdrop-blur-md bg-white/80">
        <div className="max-w-7xl mx-auto flex justify-between items-center">
          <div className="flex items-center gap-3">
            <div className="bg-indigo-600 p-2.5 rounded-2xl shadow-lg shadow-indigo-100">
              <Users className="text-white w-6 h-6" />
            </div>
            <div>
              <h1 className="text-xl font-black tracking-tight text-slate-800">2026 社團管理</h1>
              <div className="flex items-center gap-2 mt-0.5">
                <span className="w-2 h-2 bg-emerald-500 rounded-full animate-pulse"></span>
                <p className="text-[10px] text-slate-400 font-bold uppercase tracking-widest">Cloud Database Connected</p>
              </div>
            </div>
          </div>
          <div className="flex items-center gap-6">
            <div className="hidden md:block text-right border-l pl-6 border-slate-100">
              <p className="text-[10px] text-slate-400 font-bold uppercase tracking-tighter">抬頭: 天茂企管顧問有限公司</p>
              <p className="text-sm font-black text-slate-700">統編: 89177254</p>
            </div>
          </div>
        </div>
      </header>

      <main className="max-w-7xl mx-auto p-4 md:p-8">
        
        {/* Navigation Tabs */}
        <div className="flex bg-slate-200/50 p-1.5 rounded-[2rem] mb-10 w-full overflow-x-auto no-scrollbar shadow-inner">
          {[
            { id: 'accounting', label: '財務總表', icon: Wallet, color: 'text-indigo-600' },
            { id: 'courses', label: '課程報名', icon: BookOpen, color: 'text-blue-600' },
            { id: 'reimbursement', label: '經費核銷', icon: ClipboardList, color: 'text-rose-600' },
            { id: 'points', label: '積分競賽', icon: Trophy, color: 'text-amber-600' },
          ].map(tab => (
            <button 
              key={tab.id}
              onClick={() => setActiveTab(tab.id)}
              className={`flex items-center gap-2.5 px-8 py-3.5 rounded-[1.5rem] whitespace-nowrap transition-all flex-1 md:flex-none ${activeTab === tab.id ? 'bg-white shadow-xl shadow-slate-200/50 ' + tab.color + ' font-black scale-[1.02]' : 'text-slate-500 hover:text-slate-800 hover:bg-white/50'}`}
            >
              <tab.icon size={20} strokeWidth={2.5} />
              <span className="text-sm">{tab.label}</span>
            </button>
          ))}
        </div>

        {/* --- 財務總表 --- */}
        {activeTab === 'accounting' && (
          <div className="space-y-8 animate-in fade-in slide-in-from-top-4 duration-500">
            <div className="grid grid-cols-1 md:grid-cols-4 gap-6">
              <div className="bg-white p-8 rounded-[2.5rem] shadow-sm border border-slate-100 group hover:shadow-xl hover:shadow-emerald-100/50 transition-all duration-300">
                <div className="flex justify-between items-start mb-4">
                  <p className="text-slate-400 text-xs font-black uppercase tracking-widest">總收入</p>
                  <div className="bg-emerald-50 p-2 rounded-xl text-emerald-500"><TrendingUp size={20} /></div>
                </div>
                <div className="text-3xl font-black text-emerald-600 tracking-tighter">${stats.income.toLocaleString()}</div>
              </div>
              <div className="bg-white p-8 rounded-[2.5
